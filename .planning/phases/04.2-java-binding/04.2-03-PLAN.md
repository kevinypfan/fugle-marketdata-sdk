```java
package tw.com.fugle.marketdata;

import tw.com.fugle.marketdata.generated.*;
import java.util.concurrent.*;

/**
 * Fugle Market Data WebSocket client with dual streaming patterns.
 *
 * <p>Supports two streaming patterns per CONTEXT.md decisions:
 * <ul>
 *   <li>Callback-based: Implement WebSocketListener interface</li>
 *   <li>Pull-based: Use BlockingQueue with poll()/take()</li>
 * </ul>
 *
 * <p>Example (pull-based):
 * <pre>{@code
 * try (FugleWebSocketClient client = FugleWebSocketClient.builder()
 *         .apiKey(System.getenv("FUGLE_API_KEY"))
 *         .queueCapacity(1000)
 *         .build()) {
 *
 *     client.connect().join();
 *     client.subscribe("trades", "2330").join();
 *
 *     while (true) {
 *         StreamMessage msg = client.poll(5, TimeUnit.SECONDS);
 *         if (msg != null) {
 *             System.out.println(msg.getEvent());
 *         }
 *     }
 * }
 * }</pre>
 *
 * <p>Example (callback-based):
 * <pre>{@code
 * WebSocketListener listener = new WebSocketListener() {
 *     public void onMessage(StreamMessage msg) { ... }
 *     public void onConnected() { ... }
 *     public void onDisconnected() { ... }
 *     public void onError(String error) { ... }
 * };
 *
 * WebSocketClient client = MarketdataUniffiMethods.newWebSocketClient(apiKey, listener);
 * client.connect().join();
 * }</pre>
 */
public final class FugleWebSocketClient implements AutoCloseable {

    private static final int DEFAULT_QUEUE_CAPACITY = 1000;

    private final WebSocketClient inner;
    private final BlockingQueue<StreamMessage> messageQueue;
    private final BlockingQueue<String> errorQueue;
    private final InternalListener listener;

    private FugleWebSocketClient(String apiKey, int queueCapacity, WebSocketEndpoint endpoint) {
        this.messageQueue = new LinkedBlockingQueue<>(queueCapacity);
        this.errorQueue = new LinkedBlockingQueue<>(100);
        this.listener = new InternalListener(messageQueue, errorQueue);

        // Create the underlying WebSocket client with internal listener
        if (endpoint == WebSocketEndpoint.FUTOPT) {
            this.inner = MarketdataUniffiMethods.newWebSocketClientWithEndpoint(
                apiKey, listener, WebSocketEndpoint.FUTOPT);
        } else {
            this.inner = MarketdataUniffiMethods.newWebSocketClient(apiKey, listener);
        }
    }

    /**
     * Create a new builder for constructing FugleWebSocketClient.
     */
    public static Builder builder() {
        return new Builder();
    }

    /**
     * Connect to the WebSocket server.
     */
    public CompletableFuture<Void> connect() {
        return inner.connect();
    }

    /**
     * Subscribe to a channel for a symbol.
     * @param channel Channel name (e.g., "trades", "candles", "books")
     * @param symbol Symbol to subscribe (e.g., "2330")
     */
    public CompletableFuture<Void> subscribe(String channel, String symbol) {
        return inner.subscribe(channel, symbol);
    }

    /**
     * Unsubscribe from a channel for a symbol.
     */
    public CompletableFuture<Void> unsubscribe(String channel, String symbol) {
        return inner.unsubscribe(channel, symbol);
    }

    /**
     * Check if connected.
     */
    public boolean isConnected() {
        return inner.isConnected();
    }

    /**
     * Disconnect from the WebSocket server.
     */
    public CompletableFuture<Void> disconnect() {
        return inner.disconnect();
    }

    // =========================================================================
    // Pull-based API (BlockingQueue)
    // =========================================================================

    /**
     * Poll for a message with timeout.
     * @return StreamMessage or null if timeout expires
     */
    public StreamMessage poll(long timeout, TimeUnit unit) throws InterruptedException {
        return messageQueue.poll(timeout, unit);
    }

    /**
     * Take a message, blocking until available.
     */
    public StreamMessage take() throws InterruptedException {
        return messageQueue.take();
    }

    /**
     * Poll for a message without waiting.
     * @return StreamMessage or null if queue empty
     */
    public StreamMessage poll() {
        return messageQueue.poll();
    }

    /**
     * Get current queue size.
     */
    public int queueSize() {
        return messageQueue.size();
    }

    /**
     * Poll for an error with timeout.
     * @return Error message or null if no errors
     */
    public String pollError(long timeout, TimeUnit unit) throws InterruptedException {
        return errorQueue.poll(timeout, unit);
    }

    /**
     * Check if there are errors in the queue.
     */
    public boolean hasErrors() {
        return !errorQueue.isEmpty();
    }

    @Override
    public void close() {
        try {
            disconnect().join();
        } catch (Exception e) {
            // Ignore close errors
        }
    }

    // =========================================================================
    // Builder
    // =========================================================================

    public static final class Builder {
        private String apiKey;
        private int queueCapacity = DEFAULT_QUEUE_CAPACITY;
        private WebSocketEndpoint endpoint = WebSocketEndpoint.STOCK;

        private Builder() {}

        /**
         * Set API key for authentication.
         */
        public Builder apiKey(String apiKey) {
            this.apiKey = apiKey;
            return this;
        }

        /**
         * Set message queue capacity (default: 1000).
         */
        public Builder queueCapacity(int capacity) {
            this.queueCapacity = capacity;
            return this;
        }

        /**
         * Set endpoint to Stock market (default).
         */
        public Builder stock() {
            this.endpoint = WebSocketEndpoint.STOCK;
            return this;
        }

        /**
         * Set endpoint to Futures/Options market.
         */
        public Builder futopt() {
            this.endpoint = WebSocketEndpoint.FUTOPT;
            return this;
        }

        /**
         * Build the WebSocket client.
         */
        public FugleWebSocketClient build() {
            if (apiKey == null || apiKey.isEmpty()) {
                throw new IllegalStateException("apiKey is required");
            }
            return new FugleWebSocketClient(apiKey, queueCapacity, endpoint);
        }
    }

    // =========================================================================
    // Internal Listener (bridges callbacks to BlockingQueue)
    // =========================================================================

    private static class InternalListener implements WebSocketListener {
        private final BlockingQueue<StreamMessage> messageQueue;
        private final BlockingQueue<String> errorQueue;

        InternalListener(BlockingQueue<StreamMessage> messageQueue, BlockingQueue<String> errorQueue) {
            this.messageQueue = messageQueue;
            this.errorQueue = errorQueue;
        }

        @Override
        public void onMessage(StreamMessage message) {
            // Non-blocking offer - drops if queue full
            messageQueue.offer(message);
        }

        @Override
        public void onConnected() {
            // Could inject connection event as special message if needed
        }

        @Override
        public void onDisconnected() {
            // Could inject disconnection event
        }

        @Override
        public void onError(String errorMessage) {
            errorQueue.offer(errorMessage);
        }
    }
}
```

```java
        assertEquals(30L, ex.getRetryAfterSeconds());
    }

    @Test
    void rateLimitExceptionDefaultRetryAfter() {

        // Default should be 60 seconds per implementation

    }

    @Test
    void fugleExceptionFromMapsToApiException() {
        Exception apiError = new RuntimeException("API Error 500");

        java.util.concurrent.CompletionException wrapped =
```
