   ```java
   package tw.com.fugle.marketdata;

   /**
    * Base exception for all Fugle Market Data SDK errors.
    * Uses unchecked (RuntimeException) per modern Java conventions.
    */
   public class FugleException extends RuntimeException {

       public FugleException(String message) {
           super(message);
       }

       public FugleException(String message, Throwable cause) {
           super(message, cause);
       }

       /**
        * Convert a generated MarketDataException to the appropriate FugleException subtype.
        */
       public static FugleException from(Throwable t) {
           // Check for specific error types from generated code
           String className = t.getClass().getSimpleName();
           String message = t.getMessage() != null ? t.getMessage() : className;

           // Map generated exception types to hierarchy
           // UniFFI generates MarketDataException variants as nested classes or enum-like pattern
           if (className.contains("AuthError") || message.contains("Unauthorized") || message.contains("401")) {
               return new AuthException(message, t);
           } else if (className.contains("RateLimitError") || message.contains("429") || message.contains("rate limit")) {
               return new RateLimitException(message, t);
           } else if (className.contains("ApiError") || className.contains("HttpError")) {
               return new ApiException(message, t);
           }
           return new FugleException(message, t);
       }

       /**
        * Unwrap CompletionException to get the underlying FugleException.
        */
       public static FugleException unwrap(Throwable t) {
           if (t instanceof java.util.concurrent.CompletionException && t.getCause() != null) {
               return unwrap(t.getCause());
           }
           if (t instanceof FugleException) {
               return (FugleException) t;
           }
           return from(t);
       }
   }
   ```

   ```java
   import java.util.regex.Matcher;
   import java.util.regex.Pattern;

   /**
    * Exception for rate limit errors (HTTP 429).
    * Provides retry-after guidance per CONTEXT.md.
    */
   public class RateLimitException extends ApiException {
       private static final Pattern RETRY_PATTERN = Pattern.compile("retry[\\s-]?after[:\\s]*(\\d+)", Pattern.CASE_INSENSITIVE);
       private static final long DEFAULT_RETRY_SECONDS = 60L;

       private final Long retryAfterSeconds;

       public RateLimitException(String message, Throwable cause) {
           super(message, cause);
           this.retryAfterSeconds = parseRetryAfter(message);
       }

       /**
        * Get recommended seconds to wait before retrying.
        * Returns null if retry-after not available in error message.
        */
       public Long getRetryAfterSeconds() {
           return retryAfterSeconds;
       }

       private static Long parseRetryAfter(String message) {
           if (message == null) return DEFAULT_RETRY_SECONDS;
           Matcher matcher = RETRY_PATTERN.matcher(message);
           if (matcher.find()) {
               try {
                   return Long.parseLong(matcher.group(1));
               } catch (NumberFormatException e) {
                   return DEFAULT_RETRY_SECONDS;
               }
           }
           return DEFAULT_RETRY_SECONDS;
       }
   }
   ```

```java
package tw.com.fugle.marketdata;

import tw.com.fugle.marketdata.generated.*;
import java.util.concurrent.CompletableFuture;
import java.util.concurrent.CompletionException;

/**
 * Fugle Market Data REST client with builder pattern construction.
 *
 * <p>Example usage:
 * <pre>{@code
 * FugleRestClient client = FugleRestClient.builder()
 *     .apiKey(System.getenv("FUGLE_API_KEY"))
 *     .build();
 *
 * // Sync usage
 * Quote quote = client.stock().intraday().getQuote("2330");
 *
 * // Async usage
 * client.stock().intraday().getQuoteAsync("2330")
 *     .thenAccept(q -> System.out.println(q.getLastPrice()));
 * }</pre>
 */
public final class FugleRestClient implements AutoCloseable {

    private final RestClient inner;

    private FugleRestClient(RestClient inner) {
        this.inner = inner;
    }

    /**
     * Create a new builder for constructing FugleRestClient.
     */
    public static Builder builder() {
        return new Builder();
    }

    /**
     * Access stock market endpoints.
     */
    public StockClientWrapper stock() {
        return new StockClientWrapper(inner.stock());
    }

    /**
     * Access futures/options endpoints.
     */
    public FutOptClientWrapper futopt() {
        return new FutOptClientWrapper(inner.futopt());
    }

    @Override
    public void close() {
        // RestClient cleanup if needed (UniFFI handles Arc release)
    }

    /**
     * Builder for FugleRestClient supporting multiple authentication methods.
     */
    public static final class Builder {
        private String apiKey;
        private String bearerToken;
        private String sdkToken;

        private Builder() {}

        /**
         * Set API key authentication.
         */
        public Builder apiKey(String apiKey) {
            this.apiKey = apiKey;
            return this;
        }

        /**
         * Set bearer token authentication.
         */
        public Builder bearerToken(String token) {
            this.bearerToken = token;
            return this;
        }

        /**
         * Set SDK token authentication.
         */
        public Builder sdkToken(String token) {
            this.sdkToken = token;
            return this;
        }

        /**
         * Build the client with configured authentication.
         * @throws IllegalStateException if no authentication method provided
         * @throws FugleException if client creation fails
         */
        public FugleRestClient build() {
            try {
                RestClient client;
                if (sdkToken != null) {
                    client = MarketdataUniffiMethods.newRestClientWithSdkToken(sdkToken);
                } else if (apiKey != null) {
                    client = MarketdataUniffiMethods.newRestClientWithApiKey(apiKey);
                } else if (bearerToken != null) {
                    client = MarketdataUniffiMethods.newRestClientWithBearerToken(bearerToken);
                } else {
                    throw new IllegalStateException("Must provide apiKey, bearerToken, or sdkToken");
                }
                return new FugleRestClient(client);
            } catch (Exception e) {
                throw FugleException.from(e);
            }
        }
    }

    // =========================================================================
    // Stock Client Wrapper
    // =========================================================================

    public static final class StockClientWrapper {
        private final StockClient inner;

        StockClientWrapper(StockClient inner) {
            this.inner = inner;
        }

        public StockIntradayClientWrapper intraday() {
            return new StockIntradayClientWrapper(inner.intraday());
        }
    }

    public static final class StockIntradayClientWrapper {
        private final StockIntradayClient inner;

        StockIntradayClientWrapper(StockIntradayClient inner) {
            this.inner = inner;
        }

        // Sync methods (blocking)

        public Quote getQuote(String symbol) {
            try {
                return inner.quoteSync(symbol);
            } catch (Exception e) {
                throw FugleException.from(e);
            }
        }

        public Ticker getTicker(String symbol) {
            try {
                return inner.tickerSync(symbol);
            } catch (Exception e) {
                throw FugleException.from(e);
            }
        }

        public TradesResponse getTrades(String symbol) {
            try {
                return inner.tradesSync(symbol);
            } catch (Exception e) {
                throw FugleException.from(e);
            }
        }

        public IntradayCandlesResponse getCandles(String symbol, String timeframe) {
            try {
                return inner.candlesSync(symbol, timeframe);
            } catch (Exception e) {
                throw FugleException.from(e);
            }
        }

        public VolumesResponse getVolumes(String symbol) {
            try {
                return inner.volumesSync(symbol);
            } catch (Exception e) {
                throw FugleException.from(e);
            }
        }

        // Async methods (CompletableFuture)

        public CompletableFuture<Quote> getQuoteAsync(String symbol) {
            return inner.getQuote(symbol)
                .exceptionally(e -> { throw new CompletionException(FugleException.from(e)); });
        }

        public CompletableFuture<Ticker> getTickerAsync(String symbol) {
            return inner.getTicker(symbol)
                .exceptionally(e -> { throw new CompletionException(FugleException.from(e)); });
        }

        public CompletableFuture<TradesResponse> getTradesAsync(String symbol) {
            return inner.getTrades(symbol)
                .exceptionally(e -> { throw new CompletionException(FugleException.from(e)); });
        }

        public CompletableFuture<IntradayCandlesResponse> getCandlesAsync(String symbol, String timeframe) {
            return inner.getCandles(symbol, timeframe)
                .exceptionally(e -> { throw new CompletionException(FugleException.from(e)); });
        }

        public CompletableFuture<VolumesResponse> getVolumesAsync(String symbol) {
            return inner.getVolumes(symbol)
                .exceptionally(e -> { throw new CompletionException(FugleException.from(e)); });
        }
    }

    // =========================================================================
    // FutOpt Client Wrapper
    // =========================================================================

    public static final class FutOptClientWrapper {
        private final FutOptClient inner;

        FutOptClientWrapper(FutOptClient inner) {
            this.inner = inner;
        }

        public FutOptIntradayClientWrapper intraday() {
            return new FutOptIntradayClientWrapper(inner.intraday());
        }
    }

    public static final class FutOptIntradayClientWrapper {
        private final FutOptIntradayClient inner;

        FutOptIntradayClientWrapper(FutOptIntradayClient inner) {
            this.inner = inner;
        }

        public FutOptQuote getQuote(String symbol, boolean afterHours) {

            } catch (Exception e) {

            try {
                return inner.tickerSync(symbol, afterHours);

                throw FugleException.from(e);
            }
        }

        public ProductsResponse getProducts(String typ) {
            try {
                return inner.productsSync(typ);
            } catch (Exception e) {
                throw FugleException.from(e);
            }
        }

        // Async methods

        public CompletableFuture<FutOptQuote> getQuoteAsync(String symbol, boolean afterHours) {
            return inner.getQuote(symbol, afterHours)
```
