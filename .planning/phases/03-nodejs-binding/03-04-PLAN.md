const { RestClient, WebSocketClient } = require('../');

      expect(typeof intraday.quote).toBe('function');

      expect(typeof intraday.candles).toBe('function');

      expect(typeof intraday.volumes).toBe('function');

      expect(typeof intraday.volumes).toBe('function');

const { RestClient } = require('../');

const API_KEY = process.env.FUGLE_API_KEY;
const describeWithApiKey = API_KEY ? describe : describe.skip;

describeWithApiKey('REST Integration Tests', () => {
  let client;

  beforeAll(() => {
    client = new RestClient(API_KEY);
  });

  describe('Stock Intraday', () => {
    test('quote returns valid data', async () => {
      const quote = await client.stock.intraday.quote('2330');
      expect(quote).toBeDefined();
      expect(quote.symbol).toBe('2330');
      expect(typeof quote.lastPrice).toBe('number');
    });

    test('ticker returns valid data', async () => {
      const ticker = await client.stock.intraday.ticker('2330');
      expect(ticker).toBeDefined();
      expect(ticker.symbol).toBe('2330');
    });

    test('candles returns valid data', async () => {
      const candles = await client.stock.intraday.candles('2330', '5');
      expect(candles).toBeDefined();
      expect(Array.isArray(candles.data)).toBe(true);
    });
  });

  describe('Error handling', () => {
    test('invalid symbol returns error with code', async () => {
      await expect(client.stock.intraday.quote('INVALID_SYMBOL'))
        .rejects.toThrow();
    });
  });
});

// js/tests/websocket-integration.test.js
const { WebSocketClient } = require('../');

const API_KEY = process.env.FUGLE_API_KEY;
const describeWithApiKey = API_KEY ? describe : describe.skip;

describeWithApiKey('WebSocket Integration Tests', () => {
  let client;

  beforeAll(() => {
    client = new WebSocketClient(API_KEY);
  });

  afterAll(() => {
    // Cleanup any connections
    try {
      client.stock.disconnect();
      client.futopt.disconnect();
    } catch (e) {
      // Ignore cleanup errors
    }
  });

  test('stock WebSocket connects and receives messages', (done) => {
    const ws = client.stock;
    let connected = false;

    ws.on('connect', () => {
      connected = true;
      ws.subscribe({ channel: 'aggregates', symbol: '2330' });
    });

    ws.on('message', (data) => {
      expect(typeof data).toBe('string');
      const parsed = JSON.parse(data);
      expect(parsed).toBeDefined();
      ws.disconnect();
      done();
    });

    ws.on('error', (err) => {
      done.fail(err);
    });

    ws.connect();

    // Timeout after 10 seconds
    setTimeout(() => {
      if (!connected) {
        done.fail('Connection timeout');
      }
    }, 10000);
  }, 15000);
});
