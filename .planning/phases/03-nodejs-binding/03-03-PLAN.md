  intradayOddLot?: boolean;
}

  channel: 'trades' | 'books' | 'candles' | 'aggregates';
  symbol: string;
  afterHours?: boolean;

  message: (data: string) => void;  // JSON string of WebSocketMessage
  connect: (info: string) => void;
  disconnect: (reason: string) => void;
  reconnect: (info: string) => void;
  error: (error: string) => void;
}
