#[cfg_attr(feature = "js", napi_derive::napi(object))]
pub struct SmaResponse {
    pub symbol: String,
    #[serde(rename = "type")]
    pub data_type: String,
    pub exchange: String,
    pub market: String,
    pub timeframe: String,
    pub period: u32,
    pub data: Vec<SmaDataPoint>,
}
