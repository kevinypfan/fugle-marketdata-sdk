// Historical candles response (matches Node.js SDK)
#[derive(Debug, Clone, Serialize, Deserialize)]

#[cfg_attr(feature = "js", napi_derive::napi(object))]
pub struct HistoricalCandlesResponse {
    pub symbol: String,
    #[serde(rename = "type")]
    pub data_type: String,
    pub exchange: String,
    pub market: String,
    pub timeframe: Option<String>,
    #[serde(rename = "data")]
    pub candles: Vec<HistoricalCandle>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
#[cfg_attr(feature = "python", pyo3::prelude::pyclass)]
#[cfg_attr(feature = "js", napi_derive::napi(object))]
pub struct HistoricalCandle {
    pub date: String,
    pub open: f64,
    pub high: f64,
    pub low: f64,
    pub close: f64,
    pub volume: u64,
    #[serde(rename = "turnover")]
    pub turnover: Option<f64>,
    pub change: Option<f64>,
}

    pub data_type: String,
    pub exchange: String,
    pub market: String,
    // Add stat-specific fields based on official SDK
}
