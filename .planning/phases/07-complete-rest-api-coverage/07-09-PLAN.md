package fugle_test

import (
    "testing"
    "reflect"
    "github.com/fugle/marketdata-sdk/go/fugle"
)

func TestStockHistoricalClient_Exists(t *testing.T) {
    client := fugle.NewRestClient("test-key")
    historical := client.Stock().Historical()
    if historical == nil {
        t.Error("Historical client should not be nil")
    }
}

func TestStockHistoricalClient_HasCandlesMethod(t *testing.T) {
    client := fugle.NewRestClient("test-key")
    historical := client.Stock().Historical()

    typ := reflect.TypeOf(historical)
    _, found := typ.MethodByName("Candles")
    if !found {
        t.Error("Historical client should have Candles method")
    }
}

func TestStockTechnicalClient_HasIndicatorMethods(t *testing.T) {
    indicators := []string{"Sma", "Rsi", "Kdj", "Macd", "Bb"}
    client := fugle.NewRestClient("test-key")
    technical := client.Stock().Technical()

    typ := reflect.TypeOf(technical)
    for _, indicator := range indicators {
        _, found := typ.MethodByName(indicator)
        if !found {
            t.Errorf("Technical client should have %s method", indicator)
        }
    }
}
