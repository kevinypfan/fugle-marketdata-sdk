```rust
use serde::{Deserialize, Serialize};

// FutOpt Historical Candles response
#[derive(Debug, Clone, Serialize, Deserialize)]
#[cfg_attr(feature = "python", pyo3::prelude::pyclass)]
#[cfg_attr(feature = "js", napi_derive::napi(object))]
pub struct FutOptHistoricalCandlesResponse {
    pub symbol: String,
    #[serde(rename = "type")]
    pub data_type: String,
    pub exchange: String,
    pub timeframe: Option<String>,
    #[serde(rename = "data")]
    pub candles: Vec<FutOptHistoricalCandle>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
#[cfg_attr(feature = "python", pyo3::prelude::pyclass)]
#[cfg_attr(feature = "js", napi_derive::napi(object))]
pub struct FutOptHistoricalCandle {
    pub date: String,
    pub open: f64,
    pub high: f64,
    pub low: f64,
    pub close: f64,
    pub volume: u64,
    #[serde(rename = "openInterest")]
    pub open_interest: Option<u64>,
    pub change: Option<f64>,
    #[serde(rename = "changePercent")]
    pub change_percent: Option<f64>,
}

// FutOpt Daily response
#[derive(Debug, Clone, Serialize, Deserialize)]
#[cfg_attr(feature = "python", pyo3::prelude::pyclass)]
#[cfg_attr(feature = "js", napi_derive::napi(object))]
pub struct FutOptDailyResponse {
    pub symbol: String,
    #[serde(rename = "type")]
    pub data_type: String,
    pub exchange: String,
    pub data: Vec<FutOptDailyData>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
#[cfg_attr(feature = "python", pyo3::prelude::pyclass)]
#[cfg_attr(feature = "js", napi_derive::napi(object))]
pub struct FutOptDailyData {
    pub date: String,
    pub open: f64,
    pub high: f64,
    pub low: f64,
    pub close: f64,
    pub volume: u64,
    #[serde(rename = "openInterest")]
    pub open_interest: Option<u64>,
    #[serde(rename = "settlementPrice")]
    pub settlement_price: Option<f64>,
}
```
