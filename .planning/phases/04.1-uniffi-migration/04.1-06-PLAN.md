```csharp
    {
```

```go
// bindings/go/marketdata/marketdata_test.go
package marketdata

import (
    "os"
    "testing"
)

func TestNewRestClient(t *testing.T) {
    // Test client creation (doesn't require valid key)
    client, err := NewRestClient("test-api-key")
    if err != nil {
        t.Fatalf("Failed to create client: %v", err)
    }
    if client == nil {
        t.Fatal("Client is nil")
    }
}

func TestRestClient_Stock(t *testing.T) {
    client, _ := NewRestClient("test-api-key")
    stock := client.Stock()
    if stock == nil {
        t.Fatal("StockClient is nil")
    }

    intraday := stock.Intraday()
    if intraday == nil {
        t.Fatal("IntradayClient is nil")
    }
}

func TestStreamingClient_Creation(t *testing.T) {
    client, err := NewStreamingClient("test-api-key", 100)
    if err != nil {
        t.Fatalf("Failed to create streaming client: %v", err)
    }
    if client == nil {
        t.Fatal("StreamingClient is nil")
    }
}

// Integration test - requires FUGLE_API_KEY
func TestRestClient_GetQuote_Integration(t *testing.T) {
    apiKey := os.Getenv("FUGLE_API_KEY")
    if apiKey == "" {
        t.Skip("FUGLE_API_KEY not set")
    }

    client, _ := NewRestClient(apiKey)
    quote, err := client.Stock().Intraday().GetQuote("2330")
    if err != nil {
        t.Fatalf("Failed to get quote: %v", err)
    }

    if quote.Symbol != "2330" {
        t.Errorf("Expected symbol 2330, got %s", quote.Symbol)
    }
}
```

2. **Create CI workflow (.github/workflows/uniffi.yml):**
```yaml
name: UniFFI Bindings

on:
  push:
    paths:
      - 'uniffi/**'
      - 'core/**'
      - 'bindings/**'
  pull_request:
    paths:
      - 'uniffi/**'
      - 'core/**'
      - 'bindings/**'

jobs:
  build-rust:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Install Rust
        uses: dtolnay/rust-action@stable

      - name: Cache Cargo
        uses: Swatinem/rust-cache@v2
        with:
          workspaces: ". -> target"

      - name: Build UniFFI crate
        run: cargo build -p marketdata-uniffi --release

      - name: Upload library
        uses: actions/upload-artifact@v4
        with:
          name: uniffi-lib-linux
          path: target/release/libmarketdata_uniffi.so

  generate-csharp:
    needs: build-rust
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Download library
        uses: actions/download-artifact@v4
        with:
          name: uniffi-lib-linux
          path: target/release/

      - name: Install uniffi-bindgen-cs
        run: cargo install uniffi-bindgen-cs --git https://github.com/NordSecurity/uniffi-bindgen-cs --tag v0.10.0+v0.29.4

      - name: Generate C# bindings
        run: uniffi-bindgen-cs uniffi/src/marketdata.udl -o bindings/csharp/MarketdataUniffi/

      - name: Build C# project
        run: dotnet build bindings/csharp/FugleMarketData.sln

      - name: Run C# tests
        run: dotnet test bindings/csharp/FugleMarketData.Tests/ --filter "TestCategory!=Integration"

  generate-go:
    needs: build-rust
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Download library
        uses: actions/download-artifact@v4
        with:
          name: uniffi-lib-linux
          path: target/release/

      - name: Setup Go
        uses: actions/setup-go@v5
        with:
          go-version: '1.21'

      - name: Install uniffi-bindgen-go
        run: cargo install uniffi-bindgen-go --git https://github.com/NordSecurity/uniffi-bindgen-go --tag v0.2.2+v0.28.3

      - name: Generate Go bindings
        run: uniffi-bindgen-go uniffi/src/marketdata.udl -o bindings/go/marketdata/

      - name: Build Go module
        run: |
          cd bindings/go/marketdata
          CGO_ENABLED=1 go build .

      - name: Run Go tests
        run: |
          cd bindings/go/marketdata
          CGO_ENABLED=1 go test -v -short .
```
