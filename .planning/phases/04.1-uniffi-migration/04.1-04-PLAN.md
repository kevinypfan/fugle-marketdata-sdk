<task type="auto">
  <name>Task 1: Install uniffi-bindgen-cs and generate C# bindings</name>
  <files>Makefile, bindings/csharp/MarketdataUniffi/</files>
  <action>
1. **Install uniffi-bindgen-cs** (one-time, or in CI):
```bash
cargo install uniffi-bindgen-cs --git https://github.com/NordSecurity/uniffi-bindgen-cs --tag v0.10.0+v0.29.4
```

Note: Per RESEARCH.md, v0.10.0+v0.29.4 targets UniFFI 0.29.4 but should work with 0.28.x. If issues, try earlier tag.

2. **Create output directory:**
```bash
mkdir -p bindings/csharp/MarketdataUniffi
```

3. **Generate bindings:**
```bash
# First build the library to get the .so/.dylib
cargo build -p marketdata-uniffi --release

# Generate C# from UDL
uniffi-bindgen-cs uniffi/src/marketdata.udl -o bindings/csharp/MarketdataUniffi/
```

Or using cargo directly if uniffi-bindgen-cs supports it:
```bash
cargo run -p uniffi-bindgen-cs -- uniffi/src/marketdata.udl -o bindings/csharp/MarketdataUniffi/
```

4. **Add Makefile target:**
```makefile
.PHONY: gen-csharp
gen-csharp:
	cargo build -p marketdata-uniffi --release
	uniffi-bindgen-cs uniffi/src/marketdata.udl -o bindings/csharp/MarketdataUniffi/
```

5. **Verify generated file exists:**
```bash
ls bindings/csharp/MarketdataUniffi/*.cs
```

Expected: MarketdataUniffi.cs with namespace, classes for RestClient, types for Quote, etc.
  </action>

3. **Create RestClient wrapper:**
```csharp
// bindings/csharp/FugleMarketData/RestClient.cs
using MarketdataUniffi;

namespace FugleMarketData;

public class RestClient : IDisposable
{
    private readonly MarketdataUniffi.RestClient _inner;

    public RestClient(string apiKey)
    {
        _inner = MarketdataUniffi.MarketdataUniffiMethods.NewRestClient(apiKey);
    }

    public StockClient Stock => new StockClient(_inner.Stock());

    public void Dispose()
    {
        // UniFFI handles cleanup via Arc drop
    }
}

public class StockClient
{
    private readonly MarketdataUniffi.StockClient _inner;

    internal StockClient(MarketdataUniffi.StockClient inner)
    {
        _inner = inner;
    }

    public StockIntradayClient Intraday => new StockIntradayClient(_inner.Intraday());
}

public class StockIntradayClient
{
    private readonly MarketdataUniffi.StockIntradayClient _inner;

    internal StockIntradayClient(MarketdataUniffi.StockIntradayClient inner)
    {
        _inner = inner;
    }

    // Async method (matches FubonNeo pattern)
    public Task<Quote> GetQuoteAsync(string symbol)
        => _inner.GetQuoteAsync(symbol);

    // Sync wrapper if needed
    public Quote GetQuote(string symbol)
        => _inner.QuoteSync(symbol);
}
```
