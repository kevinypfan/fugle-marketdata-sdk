Create unified test workflow that runs all language binding tests:

```yaml
name: Test Suite

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

concurrency:
  group: test-${{ github.ref }}
  cancel-in-progress: true

env:
  CARGO_TERM_COLOR: always
  RUST_BACKTRACE: 1

jobs:
  # Rust core tests
  rust-tests:
    name: Rust Core Tests
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
      - uses: Swatinem/rust-cache@v2

      - name: Run unit tests
        run: cargo test -p marketdata-core --lib

      - name: Run integration tests (structural only)
        run: cargo test -p marketdata-core --test '*' -- --skip integration

  # Python binding tests
  python-tests:
    name: Python Tests
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
      - uses: Swatinem/rust-cache@v2
      - uses: actions/setup-python@v5
        with:
          python-version: '3.12'

      - name: Install Python dependencies
        run: |
          cd py
          pip install maturin
          pip install -e ".[test]"

      - name: Build native extension
        run: cd py && maturin develop

      - name: Run structural tests
        run: pytest py/tests/test_api_compatibility.py -v

      - name: Run response compatibility tests
        run: pytest py/tests/test_response_compatibility.py -v || true

      - name: Run async tests
        run: pytest py/tests/test_rest_async.py py/tests/test_websocket_async.py -v

      - name: Run GIL safety tests
        run: pytest py/tests/test_gil_safety.py -v --timeout=30

  # Python integration tests (requires API key)
  python-integration:
    name: Python Integration Tests
    runs-on: ubuntu-latest
    if: github.event_name == 'push' && github.ref == 'refs/heads/main'
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
      - uses: actions/setup-python@v5
        with:
          python-version: '3.12'

      - name: Check API key availability
        id: check-key
        run: |
          if [ -n "${{ secrets.FUGLE_API_KEY }}" ]; then
            echo "available=true" >> $GITHUB_OUTPUT
          else
            echo "available=false" >> $GITHUB_OUTPUT
          fi

      - name: Install and build
        if: steps.check-key.outputs.available == 'true'
        run: |
          cd py
          pip install maturin
          pip install -e ".[test]"
          maturin develop

      - name: Run integration tests
        if: steps.check-key.outputs.available == 'true'
        env:
          FUGLE_API_KEY: ${{ secrets.FUGLE_API_KEY }}
        run: pytest py/tests/ -m integration -v

  # Node.js binding tests
  nodejs-tests:
    name: Node.js Tests
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
      - uses: Swatinem/rust-cache@v2
      - uses: actions/setup-node@v4
        with:
          node-version: '20'

      - name: Install dependencies
        run: cd js && npm ci

      - name: Build native addon
        run: cd js && npm run build

      - name: Run structural tests
        run: cd js && npm test -- --testPathPattern=api-compatibility

      - name: Run response compatibility tests
        run: cd js && npm test -- --testPathPattern=response-compatibility || true

  # C# binding tests
  csharp-tests:
    name: C# Tests
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
      - uses: Swatinem/rust-cache@v2
      - uses: actions/setup-dotnet@v4
        with:
          dotnet-version: '8.0.x'

      - name: Build native library
        run: cargo build -p marketdata-uniffi --release

      - name: Copy native library
        run: |
          mkdir -p bindings/csharp/MarketdataUniffi.Tests/bin/Debug/net8.0
          cp target/release/libmarketdata_uniffi.so bindings/csharp/MarketdataUniffi.Tests/bin/Debug/net8.0/ || true
          cp target/release/libmarketdata_uniffi.dylib bindings/csharp/MarketdataUniffi.Tests/bin/Debug/net8.0/ || true

      - name: Run C# tests
        run: dotnet test bindings/csharp/MarketdataUniffi.Tests/

  # Java binding tests
  java-tests:
    name: Java Tests
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
      - uses: Swatinem/rust-cache@v2
      - uses: actions/setup-java@v4
        with:
          distribution: 'temurin'
          java-version: '21'

      - name: Build native library
        run: cargo build -p marketdata-uniffi --release

      - name: Copy native library
        run: |
          mkdir -p bindings/java/src/main/resources
          cp target/release/libmarketdata_uniffi.so bindings/java/src/main/resources/ || true

      - name: Run Java tests
        run: cd bindings/java && ./gradlew test

  # Go binding tests
  go-tests:
    name: Go Tests
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
      - uses: Swatinem/rust-cache@v2
      - uses: actions/setup-go@v5
        with:
          go-version: '1.22'

      - name: Build native library
        run: cargo build -p marketdata-uniffi --release

      - name: Set library path
        run: echo "LD_LIBRARY_PATH=$PWD/target/release" >> $GITHUB_ENV

      - name: Run Go tests
        run: |
          cd bindings/go/marketdata
          CGO_ENABLED=1 go test -v -short .
```

This workflow:
- Runs on every push/PR
- Tests all 5 language bindings
- Separates structural tests (always run) from integration tests (main only, with API key)
- Uses matrix for efficient parallel execution
- Handles native library availability gracefully

Create benchmark workflow using github-action-benchmark:

```yaml
name: Performance Benchmarks

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

permissions:
  contents: write
  deployments: write

jobs:
  benchmark:
    name: Run Benchmarks
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
      - uses: Swatinem/rust-cache@v2
      - uses: actions/setup-python@v5
        with:
          python-version: '3.12'

      # Build Python binding
      - name: Install Python dependencies
        run: |
          cd py
          pip install maturin
          pip install -e ".[test]"
          maturin develop --release

      # Run Rust benchmarks
      - name: Run Rust benchmarks
        run: |
          cargo bench -p marketdata-core --bench rest_latency -- --noplot --save-baseline main 2>&1 | tee rust-bench-output.txt

      # Run Python benchmarks
      - name: Run Python benchmarks
        run: |
          pytest py/tests/test_performance.py \
            --benchmark-only \
            --benchmark-json=py-bench-results.json \
            --benchmark-group-by=func \
            -v || true

      # Store benchmark results
      - name: Store Rust benchmark result
        uses: benchmark-action/github-action-benchmark@v1
        if: github.event_name == 'push'
        with:
          name: Rust Benchmarks
          tool: 'cargo'
          output-file-path: rust-bench-output.txt
          github-token: ${{ secrets.GITHUB_TOKEN }}
          auto-push: true
          alert-threshold: '200%'
          comment-on-alert: true
          fail-on-alert: false
          benchmark-data-dir-path: 'dev/bench'

      - name: Store Python benchmark result
        uses: benchmark-action/github-action-benchmark@v1
        if: github.event_name == 'push'
        with:
          name: Python Benchmarks
          tool: 'pytest'
          output-file-path: py-bench-results.json
          github-token: ${{ secrets.GITHUB_TOKEN }}
          auto-push: true
          alert-threshold: '200%'
          comment-on-alert: true
          fail-on-alert: false
          benchmark-data-dir-path: 'dev/bench'

      # PR comment with benchmark comparison
      - name: Compare benchmarks (PR only)
        uses: benchmark-action/github-action-benchmark@v1
        if: github.event_name == 'pull_request'
        with:
          name: Rust Benchmarks
          tool: 'cargo'
          output-file-path: rust-bench-output.txt
          github-token: ${{ secrets.GITHUB_TOKEN }}
          comment-always: true
          alert-threshold: '200%'
          fail-on-alert: false
          benchmark-data-dir-path: 'dev/bench'
          external-data-json-path: ./cache/benchmark-data.json
```

Configuration:
- 200% threshold for regression detection (from RESEARCH.md)
- Auto-push results to benchmark data branch
- PR comments showing comparison
- Separate tracking for Rust and Python benchmarks
