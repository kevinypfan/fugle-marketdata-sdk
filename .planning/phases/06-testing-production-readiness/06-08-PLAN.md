Alternative (simpler approach):

```bash
cd /Users/zackfan/Project/fugle/fugle-marketdata-sdk

# Create a standalone recording script
cat > /tmp/record-node-baseline.js << 'EOF'
const fs = require('fs');
const { RestClient } = require('@fugle/marketdata');

async function main() {
  const apiKey = process.env.FUGLE_API_KEY;
  if (!apiKey) { console.log('Set FUGLE_API_KEY'); process.exit(1); }

  console.log('Recording official SDK baseline...');
  const client = new RestClient({ apiKey });

  const operations = {
    quote: () => client.stock.intraday.quote({ symbol: '2330' }),
    ticker: () => client.stock.intraday.ticker({ symbol: '2330' }),
    trades: () => client.stock.intraday.trades({ symbol: '2330' }),
  };

  const results = {};

  for (const [name, op] of Object.entries(operations)) {
    console.log(`  Benchmarking ${name}...`);
    // Warmup
    for (let i = 0; i < 3; i++) { try { await op(); } catch(e) {} await sleep(500); }
    // Measure
    const latencies = [];
    for (let i = 0; i < 10; i++) {
      const start = performance.now();
      try { await op(); latencies.push(performance.now() - start); } catch(e) {}
      await sleep(500);
    }
    latencies.sort((a,b) => a - b);
    results[name] = {
      median_ms: latencies[Math.floor(latencies.length / 2)],
      mean_ms: latencies.reduce((a,b) => a+b, 0) / latencies.length,
      min_ms: Math.min(...latencies),
      max_ms: Math.max(...latencies),
      rounds: latencies.length,
    };
    console.log(`    Median: ${results[name].median_ms.toFixed(2)}ms`);
  }

  fs.writeFileSync('/Users/zackfan/Project/fugle/fugle-marketdata-sdk/js/tests/benchmarks/baseline.json',
    JSON.stringify({ sdk: '@fugle/marketdata', recorded_at: new Date().toISOString(), operations: results }, null, 2));
  console.log('Baseline written!');
}

function sleep(ms) { return new Promise(r => setTimeout(r, ms)); }
main();
EOF

# Install SDK in temp and run
cd /tmp && npm init -y && npm install @fugle/marketdata
node /tmp/record-node-baseline.js
```
