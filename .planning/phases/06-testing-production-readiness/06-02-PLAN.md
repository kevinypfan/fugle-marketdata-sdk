```javascript
/**
 * Response Compatibility Tests
 *
 * Validates that SDK responses match the structure of official @fugle/marketdata.
 * Uses nock to intercept HTTP requests and return fixture data.
 */
const nock = require('nock');
const { RestClient } = require('../');
const path = require('path');
const fs = require('fs');

const fixturesDir = path.join(__dirname, 'fixtures');

// Load fixture helper
function loadFixture(name) {
  const content = fs.readFileSync(path.join(fixturesDir, name), 'utf8');
  return JSON.parse(content);
}

describe('Response Compatibility', () => {
  let client;

  beforeAll(() => {
    client = new RestClient('test-api-key');
  });

  afterEach(() => {
    nock.cleanAll();
  });

  describe('Quote Response Structure', () => {
    test('quote response has required fields', async () => {
      const fixture = loadFixture('official-quote.json');

      nock('https://api.fugle.tw')
        .get(/\/marketdata\/v1\.0\/stock\/intraday\/quote/)
        .reply(200, fixture);

      const response = await client.stock.intraday.quote('2330');

      // Required fields from official SDK
      expect(response).toHaveProperty('symbol');
      expect(response).toHaveProperty('date');
      expect(typeof response.symbol).toBe('string');
    });

    test('quote bids/asks are arrays', async () => {
      const fixture = loadFixture('official-quote.json');

      nock('https://api.fugle.tw')
        .get(/\/marketdata\/v1\.0\/stock\/intraday\/quote/)
        .reply(200, fixture);

      const response = await client.stock.intraday.quote('2330');

      if (response.bids) {
        expect(Array.isArray(response.bids)).toBe(true);
      }
      if (response.asks) {
        expect(Array.isArray(response.asks)).toBe(true);
      }
    });
  });

  describe('Ticker Response Structure', () => {
    test('ticker response has required fields', async () => {
      const fixture = loadFixture('official-ticker.json');

      nock('https://api.fugle.tw')
        .get(/\/marketdata\/v1\.0\/stock\/intraday\/ticker/)
        .reply(200, fixture);

      const response = await client.stock.intraday.ticker('2330');

      expect(response).toHaveProperty('symbol');
      expect(response).toHaveProperty('date');
      expect(response).toHaveProperty('name');
    });
  });
});
```
