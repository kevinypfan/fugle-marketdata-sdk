"""
FFI Boundary Tests for Python Binding

Tests error handling, panic recovery, and memory safety across the
Python-Rust FFI boundary. Validates SC #3 requirements.

Run: pytest py/tests/test_ffi_boundary.py -v
"""
import pytest
import asyncio
import gc
import weakref
from concurrent.futures import ThreadPoolExecutor
import threading

from marketdata_py import RestClient, WebSocketClient
from marketdata_py import (
    MarketDataError,
    AuthError,
    ApiError,
    RateLimitError,
)


class TestErrorHandlingBoundary:
    """Test error propagation across FFI boundary."""

    def test_invalid_api_key_raises_auth_error(self):
        """Invalid API key should raise AuthError, not panic."""
        client = RestClient("invalid-key")
        # Client creation should succeed (auth happens on request)
        assert client is not None

    @pytest.mark.asyncio
    async def test_network_error_raises_api_error(self):
        """Network failures should raise ApiError with message."""
        client = RestClient("test-key")
        with pytest.raises((ApiError, MarketDataError)) as exc_info:
            await client.stock.intraday.quote("2330")

        # Error message should be meaningful, not corrupted
        assert exc_info.value.args[0] is not None
        assert len(str(exc_info.value)) > 0

    @pytest.mark.asyncio
    async def test_error_inheritance_chain(self):
        """Verify exception hierarchy: RateLimitError -> ApiError -> MarketDataError."""
        # RateLimitError should be catchable as ApiError
        assert issubclass(RateLimitError, ApiError)
        assert issubclass(ApiError, MarketDataError)
        assert issubclass(AuthError, MarketDataError)


class TestPanicRecovery:
    """Test Rust panic recovery at FFI boundary."""

    def test_client_creation_with_none_key_handled(self):
        """None API key should raise Python exception, not abort."""
        with pytest.raises((TypeError, MarketDataError)):
            RestClient(None)

    def test_client_creation_with_empty_key(self):
        """Empty API key should be handled gracefully."""
        # May succeed (auth on request) or raise clear error
        try:
            client = RestClient("")
            assert client is not None
        except MarketDataError as e:
            # Also acceptable - clear error
            assert str(e) is not None

    @pytest.mark.asyncio
    async def test_invalid_symbol_format(self):
        """Invalid symbols should raise clear error, not panic."""
        client = RestClient("test-key")
        # Very long symbol that might overflow buffers
        long_symbol = "A" * 10000
        with pytest.raises((ValueError, ApiError, MarketDataError)):
            await client.stock.intraday.quote(long_symbol)


class TestMemorySafety:
    """Test memory safety for Python-specific concerns."""

    def test_client_garbage_collection(self):
        """Client should be properly cleaned up by GC."""
        weak_refs = []

        for _ in range(100):
            client = RestClient("test-key")
            weak_refs.append(weakref.ref(client))
            del client

        gc.collect()
        gc.collect()  # Double collect for weak refs

        # Most clients should be collected
        alive = sum(1 for ref in weak_refs if ref() is not None)
        assert alive < 10, f"Memory leak: {alive} clients still alive"

    @pytest.mark.asyncio
    async def test_concurrent_access_no_corruption(self):
        """Concurrent access should not corrupt state."""
        client = RestClient("test-key")
        results = []
        errors = []

        async def make_request():
            try:
                await client.stock.intraday.quote("2330")
            except MarketDataError as e:
                errors.append(str(e))
            except Exception as e:
                # Unexpected error - potential corruption
                results.append(f"UNEXPECTED: {type(e).__name__}: {e}")

        # Run 10 concurrent requests
        await asyncio.gather(*[make_request() for _ in range(10)])

        # All errors should be consistent (same type)
        if errors:
            # No memory corruption - all errors are parseable strings
            for err in errors:
                assert isinstance(err, str)

    def test_websocket_callback_exception_handling(self):
        """Exceptions in Python callbacks should not crash Rust."""
        client = WebSocketClient("test-key")

        def bad_callback(msg):
            raise ValueError("Callback exception")

        # Register callback that raises
        client.on("message", bad_callback)

        # Client should still be usable (exception logged, not propagated to Rust)
        assert client is not None


class TestGilSafety:
    """Test GIL (Global Interpreter Lock) handling."""

    @pytest.mark.timeout(10)
    def test_blocking_call_releases_gil(self):
        """Blocking calls should release GIL for other threads."""
        client = RestClient("test-key")
        other_thread_ran = threading.Event()

        def other_work():
            # This should run while main thread is in FFI call
            other_thread_ran.set()

        # Start other thread
        thread = threading.Thread(target=other_work)
        thread.start()

        # Make FFI call (will fail but that's ok)
        try:
            loop = asyncio.new_event_loop()
            loop.run_until_complete(client.stock.intraday.quote("2330"))
        except MarketDataError:
            pass  # Expected
        finally:
            loop.close()

        thread.join(timeout=5)
        assert other_thread_ran.is_set(), "GIL was not released during FFI call"

    @pytest.mark.timeout(15)
    @pytest.mark.asyncio
    async def test_async_operations_release_gil(self):
        """Async operations should not block other async tasks."""
        client = RestClient("test-key")

        async def other_task():
            await asyncio.sleep(0.1)
            return "completed"

        async def ffi_task():
            try:
                await client.stock.intraday.quote("2330")
            except MarketDataError:
                pass

        # Both should complete without timeout
        results = await asyncio.gather(
            other_task(),
            ffi_task(),
            return_exceptions=True
        )

        assert results[0] == "completed", "Other task was blocked by FFI call"

/**
 * FFI Boundary Tests for Node.js Binding
 *
 * Tests error handling, panic recovery, and memory safety across
 * the Node.js-Rust FFI boundary via napi-rs.
 *
 * Run: npm test -- --testPathPattern=ffi-boundary
 */

const { RestClient, WebSocketClient } = require('../');

describe('FFI Boundary Tests', () => {
  describe('Error Handling Boundary', () => {
    test('invalid API key creates client without crashing', () => {
      // Client creation should succeed (auth on request)
      const client = new RestClient('invalid-key');
      expect(client).toBeDefined();
    });

    test('API error propagates as JavaScript Error', async () => {
      const client = new RestClient('test-key');

      await expect(client.stock.intraday.quote('2330'))
        .rejects
        .toThrow();
    });

    test('error message is readable string', async () => {
      const client = new RestClient('test-key');

      try {
        await client.stock.intraday.quote('2330');
        fail('Should have thrown');
      } catch (error) {
        // Error message should be meaningful, not corrupted memory
        expect(typeof error.message).toBe('string');
        expect(error.message.length).toBeGreaterThan(0);
      }
    });

    test('error has proper stack trace', async () => {
      const client = new RestClient('test-key');

      try {
        await client.stock.intraday.quote('2330');
        fail('Should have thrown');
      } catch (error) {
        expect(error.stack).toBeDefined();
        expect(typeof error.stack).toBe('string');
      }
    });
  });

  describe('Panic Recovery', () => {
    test('empty API key handled gracefully', () => {
      // Should not crash the process
      expect(() => new RestClient('')).not.toThrow();
    });

    test('undefined API key throws TypeError', () => {
      expect(() => new RestClient(undefined)).toThrow();
    });

    test('null API key throws TypeError', () => {
      expect(() => new RestClient(null)).toThrow();
    });

    test('very long symbol does not cause buffer overflow', async () => {
      const client = new RestClient('test-key');
      const longSymbol = 'A'.repeat(10000);

      // Should throw error, not crash
      await expect(client.stock.intraday.quote(longSymbol))
        .rejects
        .toThrow();
    });
  });

  describe('Memory Safety', () => {
    test('client can be garbage collected', async () => {
      // Create and discard many clients
      for (let i = 0; i < 100; i++) {
        const client = new RestClient('test-key');
        // Let it go out of scope
      }

      // Force GC if available (--expose-gc flag)
      if (global.gc) {
        global.gc();
      }

      // If we got here without crash, memory is being managed
      expect(true).toBe(true);
    });

    test('concurrent requests do not corrupt state', async () => {
      const client = new RestClient('test-key');

      // Launch 10 concurrent requests
      const promises = Array(10).fill(null).map(() =>
        client.stock.intraday.quote('2330').catch(e => e)
      );

      const results = await Promise.all(promises);

      // All should be consistent errors (same type)
      for (const result of results) {
        expect(result).toBeInstanceOf(Error);
        expect(typeof result.message).toBe('string');
      }
    });

    test('response buffers are properly converted', async () => {
      const client = new RestClient('test-key');

      try {
        const result = await client.stock.intraday.quote('2330');
        // If we get here, verify result is plain JS object
        if (result) {
          expect(typeof result).toBe('object');
          // No ArrayBuffer or TypedArray should leak
          expect(result).not.toBeInstanceOf(ArrayBuffer);
        }
      } catch (error) {
        // Expected without valid API key
        expect(error).toBeInstanceOf(Error);
      }
    });
  });

  describe('Event Loop Safety', () => {
    test('async operations do not block event loop', async () => {
      const client = new RestClient('test-key');
      let otherTaskRan = false;

      // Schedule immediate task
      const immediatePromise = new Promise(resolve => {
        setImmediate(() => {
          otherTaskRan = true;
          resolve();
        });
      });

      // Start FFI call
      const ffiPromise = client.stock.intraday.quote('2330').catch(() => {});

      // Wait for immediate
      await immediatePromise;
      await ffiPromise;

      expect(otherTaskRan).toBe(true);
    });

    test('ThreadsafeFunction callback errors do not crash', () => {
      const client = new WebSocketClient('test-key');

      // Register callback that throws
      client.on('message', () => {
        throw new Error('Callback error');
      });

      // Client should still be valid
      expect(client).toBeDefined();
    });
  });
});

using Microsoft.VisualStudio.TestTools.UnitTesting;
using System;
using System.Threading.Tasks;
using System.Collections.Generic;

namespace MarketdataUniffi.Tests;

/// <summary>
/// FFI Boundary Tests for C# Binding
///
/// Tests error handling, panic recovery, and memory safety across
/// the C#-Rust FFI boundary via UniFFI.
/// </summary>
[TestClass]
public class FfiBoundaryTests
{
    private static bool _nativeLibraryAvailable;

    [ClassInitialize]
    public static void ClassInit(TestContext context)
    {
        try
        {
            using var client = new FugleMarketData.RestClient("test-key");
            _nativeLibraryAvailable = true;
        }
        catch (DllNotFoundException) { _nativeLibraryAvailable = false; }
        catch (TypeInitializationException ex) when (ex.InnerException is DllNotFoundException)
        { _nativeLibraryAvailable = false; }
        catch { _nativeLibraryAvailable = true; }
    }

    private void SkipIfNativeLibraryUnavailable()
    {
        if (!_nativeLibraryAvailable)
            Assert.Inconclusive("Native library not available");
    }

    // ========== Error Handling Boundary ==========

    [TestMethod]
    public void InvalidApiKey_CreatesClientWithoutCrash()
    {
        SkipIfNativeLibraryUnavailable();

        // Client creation should succeed (auth happens on request)
        using var client = new FugleMarketData.RestClient("invalid-key");
        Assert.IsNotNull(client);
    }

    [TestMethod]
    public async Task ApiError_PropagatesAsException()
    {
        SkipIfNativeLibraryUnavailable();

        using var client = new FugleMarketData.RestClient("test-key");

        var exception = await Assert.ThrowsExceptionAsync<Exception>(async () =>
        {
            await client.Stock.Intraday.GetQuoteAsync("2330");
        });

        // Error message should be readable, not corrupted memory
        Assert.IsNotNull(exception.Message);
        Assert.IsTrue(exception.Message.Length > 0);
    }

    [TestMethod]
    public async Task ErrorMessage_IsReadableString()
    {
        SkipIfNativeLibraryUnavailable();

        using var client = new FugleMarketData.RestClient("test-key");

        try
        {
            await client.Stock.Intraday.GetQuoteAsync("2330");
            Assert.Fail("Should have thrown");
        }
        catch (Exception ex)
        {
            // Verify message is not corrupted
            Assert.IsFalse(string.IsNullOrEmpty(ex.Message));
            // Message should be valid UTF-8 (no encoding issues)
            var bytes = System.Text.Encoding.UTF8.GetBytes(ex.Message);
            var roundtrip = System.Text.Encoding.UTF8.GetString(bytes);
            Assert.AreEqual(ex.Message, roundtrip);
        }
    }

    // ========== Panic Recovery ==========

    [TestMethod]
    public void EmptyApiKey_HandledGracefully()
    {
        SkipIfNativeLibraryUnavailable();

        // Should not crash the process
        try
        {
            using var client = new FugleMarketData.RestClient("");
            Assert.IsNotNull(client);
        }
        catch (Exception ex)
        {
            // Also acceptable - clear exception
            Assert.IsNotNull(ex.Message);
        }
    }

    [TestMethod]
    public async Task VeryLongSymbol_DoesNotCauseBufferOverflow()
    {
        SkipIfNativeLibraryUnavailable();

        using var client = new FugleMarketData.RestClient("test-key");
        var longSymbol = new string('A', 10000);

        // Should throw error, not crash or corrupt memory
        await Assert.ThrowsExceptionAsync<Exception>(async () =>
        {
            await client.Stock.Intraday.GetQuoteAsync(longSymbol);
        });
    }

    // ========== Memory Safety ==========

    [TestMethod]
    public void ClientDispose_ReleasesNativeResources()
    {
        SkipIfNativeLibraryUnavailable();

        var weakRef = CreateAndDisposeClient();

        // Force GC
        GC.Collect();
        GC.WaitForPendingFinalizers();
        GC.Collect();

        // Client should be collected
        Assert.IsFalse(weakRef.IsAlive, "Client was not garbage collected");
    }

    private WeakReference CreateAndDisposeClient()
    {
        var client = new FugleMarketData.RestClient("test-key");
        var weakRef = new WeakReference(client);
        client.Dispose();
        return weakRef;
    }

    [TestMethod]
    public async Task ConcurrentRequests_DoNotCorruptState()
    {
        SkipIfNativeLibraryUnavailable();

        using var client = new FugleMarketData.RestClient("test-key");
        var errors = new List<string>();
        var tasks = new List<Task>();

        for (int i = 0; i < 10; i++)
        {
            tasks.Add(Task.Run(async () =>
            {
                try
                {
                    await client.Stock.Intraday.GetQuoteAsync("2330");
                }
                catch (Exception ex)
                {
                    lock (errors)
                    {
                        errors.Add(ex.Message);
                    }
                }
            }));
        }

        await Task.WhenAll(tasks);

        // All errors should be valid strings (no corruption)
        foreach (var error in errors)
        {
            Assert.IsFalse(string.IsNullOrEmpty(error));
        }
    }

    [TestMethod]
    public void MultipleClientInstances_IndependentLifetimes()
    {
        SkipIfNativeLibraryUnavailable();

        var clients = new List<FugleMarketData.RestClient>();

        // Create multiple clients
        for (int i = 0; i < 10; i++)
        {
            clients.Add(new FugleMarketData.RestClient($"key-{i}"));
        }

        // Dispose in random order
        clients[5].Dispose();
        clients[2].Dispose();
        clients[8].Dispose();

        // Remaining clients should still work
        Assert.IsNotNull(clients[0]);
        Assert.IsNotNull(clients[3]);

        // Cleanup
        foreach (var client in clients)
        {
            client.Dispose();
        }
    }

    // ========== Thread Safety ==========

    [TestMethod]
    public async Task AsyncOperations_DoNotBlockThreadPool()
    {
        SkipIfNativeLibraryUnavailable();

        using var client = new FugleMarketData.RestClient("test-key");
        bool otherTaskRan = false;

        var ffiTask = Task.Run(async () =>
        {
            try
            {
                await client.Stock.Intraday.GetQuoteAsync("2330");
            }
            catch { }
        });

        var otherTask = Task.Run(() =>
        {
            Thread.Sleep(100);
            otherTaskRan = true;
        });

        await Task.WhenAll(ffiTask, otherTask);

        Assert.IsTrue(otherTaskRan, "FFI call blocked thread pool");
    }
}
