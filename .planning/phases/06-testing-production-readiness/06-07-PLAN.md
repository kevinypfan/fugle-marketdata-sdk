```bash
# 2. Create recording script
cat > record-fixtures.js << 'EOF'
const fs = require('fs');
const { RestClient } = require('@fugle/marketdata');

async function main() {
  const apiKey = process.env.FUGLE_API_KEY;
  if (!apiKey) {
    console.log('Set FUGLE_API_KEY first');
    process.exit(1);
  }

  const client = new RestClient({ apiKey });

  const fixtures = {
    quote: await client.stock.intraday.quote({ symbol: '2330' }),
    ticker: await client.stock.intraday.ticker({ symbol: '2330' }),
    trades: await client.stock.intraday.trades({ symbol: '2330' }),
  };

  for (const [name, data] of Object.entries(fixtures)) {
    const path = `/Users/zackfan/Project/fugle/fugle-marketdata-sdk/js/tests/fixtures/official_sdk_${name}.json`;
    fs.writeFileSync(path, JSON.stringify(data, null, 2));
    console.log(`Wrote: ${path}`);
  }
}

main().catch(console.error);
EOF

# 3. Run with API key
export FUGLE_API_KEY="your-key-here"
node record-fixtures.js
```

This will create JSON fixture files:
- js/tests/fixtures/official_sdk_quote.json
