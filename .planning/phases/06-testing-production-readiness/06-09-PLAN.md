cd /Users/zackfan/Project/fugle/fugle-marketdata-sdk/py
python -c "
import asyncio
from marketdata_py import WebSocketClient

received_messages = []

def on_message(msg):
    received_messages.append(msg)
    print(f'Received: {msg[:100]}...' if len(msg) > 100 else f'Received: {msg}')

async def test_ws():
    client = WebSocketClient()
    client.on_message(on_message)

    await client.connect('wss://api.fugle.tw/marketdata/v1.0/stock/streaming')
    await client.subscribe('2330')

    print('Waiting for messages (10 seconds)...')
    await asyncio.sleep(10)

    print(f'Total messages received: {len(received_messages)}')
    return len(received_messages) > 0

success = asyncio.run(test_ws())
print('WebSocket test:', 'PASS' if success else 'FAIL')
"

# Node.js WebSocket test
cd /Users/zackfan/Project/fugle/fugle-marketdata-sdk/js
node -e "
const { WebSocketClient } = require('./index.js');

async function test() {
  const client = new WebSocketClient({ apiKey: process.env.FUGLE_API_KEY });
  let messageCount = 0;

  client.on('message', (msg) => {
    messageCount++;
    console.log('Received:', JSON.stringify(msg).slice(0, 100) + '...');
  });

  await client.connect();
  await client.subscribe('2330');

  console.log('Waiting for messages (10 seconds)...');
  await new Promise(r => setTimeout(r, 10000));

  console.log('Total messages:', messageCount);
  console.log('WebSocket test:', messageCount > 0 ? 'PASS' : 'FAIL');

  client.close();
}

test().catch(console.error);
"
