       """Test REST client instantiation."""

           client = RestClient(mock_api_key)
           assert client is not None

       def test_create_with_bearer_token(self):
           client = RestClient.with_bearer_token("test-token")
           assert client is not None

       def test_create_with_sdk_token(self):
           client = RestClient.with_sdk_token("test-sdk-token")
           assert client is not None

       def test_stock_property_returns_client(self, mock_api_key):
           client = RestClient(mock_api_key)
           assert client.stock is not None
           assert client.stock.intraday is not None

   class TestAsyncMethods:
       """Test that methods return awaitables."""

       @pytest.mark.asyncio
       async def test_quote_returns_awaitable(self, mock_api_key):
           client = RestClient(mock_api_key)
           # This should raise an error (invalid key) but BE awaitable
           with pytest.raises((MarketDataError, AuthError)):
               await client.stock.intraday.quote("2330")

   @pytest.mark.integration
   class TestIntegrationRest:
       """Integration tests requiring real API key."""

       @pytest.mark.asyncio
       async def test_quote_returns_dict(self, rest_client):
           result = await rest_client.stock.intraday.quote("2330")
           assert isinstance(result, dict)
           assert "symbol" in result or "data" in result

       @pytest.mark.asyncio
       async def test_ticker_returns_dict(self, rest_client):
           result = await rest_client.stock.intraday.ticker("2330")
           assert isinstance(result, dict)

       @pytest.mark.asyncio
       async def test_invalid_symbol_raises_error(self, rest_client):
           with pytest.raises(MarketDataError):
               await rest_client.stock.intraday.quote("INVALID_SYMBOL_12345")

2. Create py/tests/test_websocket_async.py:
   ```python
   """Tests for async WebSocket client."""
   import asyncio
   import pytest
   from marketdata_py import (
       WebSocketClient,
       MarketDataError,
       ConnectionError,
   )

   class TestWebSocketClientCreation:
       """Test WebSocket client instantiation."""

       def test_create_with_api_key(self, mock_api_key):
           client = WebSocketClient(mock_api_key)
           assert client is not None

       def test_stock_property_returns_client(self, mock_api_key):
           client = WebSocketClient(mock_api_key)
           assert client.stock is not None

       def test_futopt_property_returns_client(self, mock_api_key):
           client = WebSocketClient(mock_api_key)
           assert client.futopt is not None

   class TestAsyncConnect:
       """Test async connect method."""

       @pytest.mark.asyncio
       async def test_connect_returns_awaitable(self, mock_api_key):
           client = WebSocketClient(mock_api_key)
           # Should fail with auth error but be awaitable
           with pytest.raises((MarketDataError, ConnectionError)):
               await client.stock.connect()

   class TestCallbackPattern:
       """Test callback registration."""

       def test_on_registers_callback(self, mock_api_key):
           client = WebSocketClient(mock_api_key)
           called = []

           def handler(msg):
               called.append(msg)

           client.stock.on("message", handler)
           # No error means callback registered

       def test_off_removes_callback(self, mock_api_key):
           client = WebSocketClient(mock_api_key)
           client.stock.on("message", lambda m: None)
           client.stock.off("message")
           # No error means callback removed

   @pytest.mark.integration
   class TestIntegrationWebSocket:
       """Integration tests requiring real API key."""

       @pytest.mark.asyncio
       @pytest.mark.timeout(10)
       async def test_connect_and_subscribe(self, ws_client):
           await ws_client.stock.connect()
           assert ws_client.stock.is_connected()

           sub_id = await ws_client.stock.subscribe("trades", "2330")
           assert sub_id is not None

           await ws_client.stock.disconnect()

       @pytest.mark.asyncio
       @pytest.mark.timeout(15)
       async def test_async_iterator_receives_messages(self, ws_client):
           await ws_client.stock.connect()
           await ws_client.stock.subscribe("trades", "2330")

           messages = []
           async for msg in ws_client.stock.messages(timeout_ms=5000):
               messages.append(msg)
               if len(messages) >= 1:
                   break

           await ws_client.stock.disconnect()
           # May or may not receive messages depending on market hours
           # Just verify no crash

       @pytest.mark.asyncio
       async def test_context_manager(self, api_key):
           ws = WebSocketClient(api_key)
           async with ws.stock:
               assert ws.stock.is_connected()
           # After context exit, should be disconnected
   ```
