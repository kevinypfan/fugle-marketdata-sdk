3. Convert quote() method:
   - Change return type from `PyResult<Py<PyDict>>` to `PyResult<Bound<'py, PyAny>>`
   - Use future_into_py() to wrap the operation
   - Use tokio::task::spawn_blocking for the sync HTTP call
   - Handle errors with proper Python exception conversion

   Pattern:
   ```rust
   pub fn quote<'py>(&self, py: Python<'py>, symbol: String, odd_lot: bool) -> PyResult<Bound<'py, PyAny>> {
       let client = self.inner.clone();
       future_into_py(py, async move {
           // Wrap blocking ureq call in spawn_blocking
           let result = tokio::task::spawn_blocking(move || {
               let stock = client.stock();
               let intraday = stock.intraday();
               let mut builder = intraday.quote().symbol(&symbol);
               if odd_lot {
                   builder = builder.odd_lot(true);
               }
               builder.send()
           }).await
               .map_err(|e| pyo3::exceptions::PyRuntimeError::new_err(format!("Task join error: {}", e)))?;

           match result {
               Ok(quote) => Python::with_gil(|py| crate::types::quote_to_py(py, &quote)),
               Err(e) => Err(crate::errors::to_py_err(e)),
           }
       })
   }
   ```
