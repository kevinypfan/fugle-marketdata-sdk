Create `.github/workflows/ci.yml` that:

1. Triggers on push to main and pull_request
2. Uses dorny/paths-filter to detect which parts changed
3. Calls language-specific workflows based on changes

```yaml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  changes:
    name: Detect Changes
    runs-on: ubuntu-latest
    outputs:
      core: ${{ steps.filter.outputs.core }}
      python: ${{ steps.filter.outputs.python }}
      nodejs: ${{ steps.filter.outputs.nodejs }}
      csharp: ${{ steps.filter.outputs.csharp }}
    steps:
      - uses: actions/checkout@v4
      - uses: dorny/paths-filter@v3
        id: filter
        with:
          filters: |
            core:
              - 'core/**'
              - 'Cargo.toml'
              - 'Cargo.lock'
            python:
              - 'py/**'
              - 'core/**'
              - 'Cargo.toml'
              - 'Cargo.lock'
            nodejs:
              - 'js/**'
              - 'core/**'
              - 'Cargo.toml'
              - 'Cargo.lock'
            csharp:
              - 'uniffi/**'
              - 'core/**'
              - 'Cargo.toml'
              - 'Cargo.lock'

  python:
    name: Python Bindings
    needs: changes
    if: ${{ needs.changes.outputs.python == 'true' }}
    uses: ./.github/workflows/python.yml

  nodejs:
    name: Node.js Bindings
    needs: changes
    if: ${{ needs.changes.outputs.nodejs == 'true' }}
    uses: ./.github/workflows/nodejs.yml

  csharp:
    name: C# Bindings
    needs: changes
    if: ${{ needs.changes.outputs.csharp == 'true' }}
    uses: ./.github/workflows/csharp.yml
```

Ensure the directory `.github/workflows/` is created first.

  <name>Task 2: Create Python binding workflow</name>
  <files>.github/workflows/python.yml</files>
  <action>
Create `.github/workflows/python.yml`:

```yaml
name: Python Bindings

on:
  workflow_call:
  workflow_dispatch:

jobs:
  test:
    name: Test Python (${{ matrix.os }})
    strategy:
      fail-fast: false
      matrix:
        os: [ubuntu-latest, macos-latest, windows-latest]
        python-version: ['3.8', '3.11', '3.12']
        exclude:
          # Reduce matrix size - test 3.8 only on Linux, 3.11/3.12 on all
          - os: macos-latest
            python-version: '3.8'
          - os: windows-latest
            python-version: '3.8'
    runs-on: ${{ matrix.os }}

    steps:
      - uses: actions/checkout@v4

      - name: Install Rust toolchain
        uses: dtolnay/rust-toolchain@stable

      - name: Cache Rust dependencies
        uses: Swatinem/rust-cache@v2
        with:
          workspaces: py -> target
          shared-key: python-${{ matrix.os }}

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: ${{ matrix.python-version }}

      - name: Install maturin
        run: pip install maturin pytest

      - name: Build Python binding
        working-directory: py
        run: maturin develop

      - name: Run Python tests
        working-directory: py
        run: |
          python -c "import marketdata_py; print('Import successful')"
        # pytest tests/ will be added in Phase 2
```

Key points:
- workflow_call allows ci.yml to invoke this
- workflow_dispatch allows manual runs
- Matrix tests Python 3.8 (min) and 3.11/3.12 (current) across platforms
- Swatinem/rust-cache with workspaces config for proper caching
