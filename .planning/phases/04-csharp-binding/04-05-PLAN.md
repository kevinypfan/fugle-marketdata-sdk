3. Create WebSocketClient.cs with EventHandler pattern:
   ```csharp
   using System.Runtime.InteropServices;

   namespace Fugle.MarketData;

   /// <summary>
   /// WebSocket client for real-time market data streaming.
   /// Uses EventHandler pattern for message delivery.
   /// </summary>
   public sealed class WebSocketClient : IAsyncDisposable, IDisposable
   {
       private IntPtr _handle;
       private bool _disposed;
       private CancellationTokenSource? _pollCts;
       private Task? _pollTask;
       private readonly string _apiKey;

       /// <summary>
       /// Raised when a message is received.
       /// </summary>
       public event EventHandler<MessageEventArgs>? MessageReceived;

       /// <summary>
       /// Raised when connected to server.
       /// </summary>
       public event EventHandler<EventArgs>? Connected;

       /// <summary>
       /// Raised when disconnected from server.
       /// </summary>
       public event EventHandler<EventArgs>? Disconnected;

       /// <summary>
       /// Raised when an error occurs.
       /// </summary>
       public event EventHandler<ErrorEventArgs>? Error;

       /// <summary>
       /// Current connection state.
       /// </summary>
       public ConnectionState State
       {
           get
           {
               if (_handle == IntPtr.Zero) return ConnectionState.Disconnected;
               var code = NativeMethods.fugle_ws_get_state(_handle);
               return (ConnectionState)code;
           }
       }

       public WebSocketClient(string apiKey)
       {
           if (string.IsNullOrEmpty(apiKey))
               throw new ArgumentNullException(nameof(apiKey));

           _apiKey = apiKey;
           _handle = NativeMethods.fugle_ws_client_new(apiKey);
           if (_handle == IntPtr.Zero)
               throw new FugleException("Failed to create WebSocket client");
       }

       /// <summary>
       /// Connect to the WebSocket server.
       /// </summary>
       public async Task ConnectAsync(CancellationToken cancellationToken = default)
       {
           ThrowIfDisposed();

           var result = await Task.Run(() =>
               NativeMethods.fugle_ws_connect(_handle, _apiKey),
               cancellationToken
           ).ConfigureAwait(false);

           if (result != NativeErrorCodes.Success)
               throw NativeErrorCodes.ToException(result);

           // Start background polling
           _pollCts = new CancellationTokenSource();
           _pollTask = Task.Run(() => PollLoop(_pollCts.Token), _pollCts.Token);

           Connected?.Invoke(this, EventArgs.Empty);
       }

       /// <summary>
       /// Connect synchronously.
       /// </summary>
       public void Connect()
           => ConnectAsync().GetAwaiter().GetResult();

       /// <summary>
       /// Disconnect from the WebSocket server.
       /// </summary>
       public async Task DisconnectAsync(CancellationToken cancellationToken = default)
       {
           ThrowIfDisposed();

           // Stop polling first
           _pollCts?.Cancel();
           if (_pollTask != null)
           {
               try { await _pollTask.ConfigureAwait(false); }
               catch (OperationCanceledException) { }
           }

           var result = await Task.Run(() =>
               NativeMethods.fugle_ws_disconnect(_handle),
               cancellationToken
           ).ConfigureAwait(false);

           Disconnected?.Invoke(this, EventArgs.Empty);
       }

       /// <summary>
       /// Disconnect synchronously.
       /// </summary>
       public void Disconnect()
           => DisconnectAsync().GetAwaiter().GetResult();

       #region Subscriptions

       /// <summary>
       /// Subscribe to a stock channel.
       /// </summary>
       public Task SubscribeStockAsync(string channel, string symbol, CancellationToken cancellationToken = default)
       {
           ThrowIfDisposed();
           return Task.Run(() =>
           {
               var result = NativeMethods.fugle_ws_stock_subscribe(_handle, channel, symbol);
               if (result != NativeErrorCodes.Success)
                   throw NativeErrorCodes.ToException(result);
           }, cancellationToken);
       }

       /// <summary>
       /// Subscribe to a stock channel synchronously.
       /// </summary>
       public void SubscribeStock(string channel, string symbol)
           => SubscribeStockAsync(channel, symbol).GetAwaiter().GetResult();

       /// <summary>
       /// Unsubscribe from a stock channel.
       /// </summary>
       public Task UnsubscribeStockAsync(string channel, string symbol, CancellationToken cancellationToken = default)
       {
           ThrowIfDisposed();
           return Task.Run(() =>
           {
               var result = NativeMethods.fugle_ws_stock_unsubscribe(_handle, channel, symbol);
               if (result != NativeErrorCodes.Success)
                   throw NativeErrorCodes.ToException(result);
           }, cancellationToken);
       }

       /// <summary>
       /// Subscribe to a FutOpt channel.
       /// </summary>
       public Task SubscribeFutOptAsync(string channel, string symbol, CancellationToken cancellationToken = default)
       {
           ThrowIfDisposed();
           return Task.Run(() =>
           {
               var result = NativeMethods.fugle_ws_futopt_subscribe(_handle, channel, symbol);
               if (result != NativeErrorCodes.Success)
                   throw NativeErrorCodes.ToException(result);
           }, cancellationToken);
       }

       /// <summary>
       /// Unsubscribe from a FutOpt channel.
       /// </summary>
       public Task UnsubscribeFutOptAsync(string channel, string symbol, CancellationToken cancellationToken = default)
       {
           ThrowIfDisposed();
           return Task.Run(() =>
           {
               var result = NativeMethods.fugle_ws_futopt_unsubscribe(_handle, channel, symbol);
               if (result != NativeErrorCodes.Success)
                   throw NativeErrorCodes.ToException(result);
           }, cancellationToken);
       }

       #endregion

       #region Polling

       private async Task PollLoop(CancellationToken cancellationToken)
       {
           const int pollIntervalMs = 10; // 10ms for low latency

           while (!cancellationToken.IsCancellationRequested)
           {
               try
               {
                   IntPtr messagePtr;
                   var code = NativeMethods.fugle_ws_poll_message(_handle, out messagePtr);

                   if (code == 1) // MESSAGE_AVAILABLE
                   {
                       var json = Marshal.PtrToStringUTF8(messagePtr);
                       NativeMethods.fugle_free_string(messagePtr);

                       if (!string.IsNullOrEmpty(json))
                       {
                           MessageReceived?.Invoke(this, new MessageEventArgs(json));
                       }
                   }
                   else if (code == 0) // NO_MESSAGE
                   {
                       await Task.Delay(pollIntervalMs, cancellationToken).ConfigureAwait(false);
                   }
                   else // Error
                   {
                       Error?.Invoke(this, new ErrorEventArgs(NativeErrorCodes.ToException(code)));
                       await Task.Delay(100, cancellationToken).ConfigureAwait(false); // Back off on error
                   }
               }
               catch (OperationCanceledException)
               {
                   break;
               }
               catch (Exception ex)
               {
                   Error?.Invoke(this, new ErrorEventArgs(ex));
                   await Task.Delay(100, cancellationToken).ConfigureAwait(false);
               }
           }
       }

       #endregion

       #region IDisposable

       private void ThrowIfDisposed()
       {
           if (_disposed)
               throw new ObjectDisposedException(nameof(WebSocketClient));
       }

       public async ValueTask DisposeAsync()
       {
           if (_disposed) return;

           // Stop polling
           _pollCts?.Cancel();
           if (_pollTask != null)
           {
               try { await _pollTask.ConfigureAwait(false); }
               catch (OperationCanceledException) { }
           }

           Dispose(disposing: true);
           GC.SuppressFinalize(this);
       }

       public void Dispose()
       {
           Dispose(disposing: true);
           GC.SuppressFinalize(this);
       }

       private void Dispose(bool disposing)
       {
           if (_disposed) return;

           if (disposing)
           {
               _pollCts?.Dispose();
           }

           if (_handle != IntPtr.Zero)
           {
               NativeMethods.fugle_ws_client_free(_handle);
               _handle = IntPtr.Zero;
           }

           _disposed = true;
       }

       ~WebSocketClient() => Dispose(disposing: false);

       #endregion
   }
   ```
