   ```csharp
   namespace Fugle.MarketData;

   /// <summary>
   /// Base exception for all Fugle Market Data errors.
   /// </summary>
   public class FugleException : Exception
   {
       public int ErrorCode { get; }

       public FugleException(string message, int errorCode = 0)
           : base(message)
       {
           ErrorCode = errorCode;
       }

       public FugleException(string message, int errorCode, Exception inner)
           : base(message, inner)
       {
           ErrorCode = errorCode;
       }
   }

   /// <summary>
   /// API request failed (HTTP 4xx/5xx errors).
   /// </summary>
   public class ApiException : FugleException
   {
       public int? StatusCode { get; }

       public ApiException(string message, int errorCode, int? statusCode = null)
           : base(message, errorCode)
       {
           StatusCode = statusCode;
       }
   }

   /// <summary>
   /// Authentication failed (invalid API key, expired token).
   /// </summary>
   public class AuthException : FugleException
   {
       public AuthException(string message, int errorCode = -2)
           : base(message, errorCode) { }
   }

   /// <summary>
   /// Rate limit exceeded. Check RetryAfter for when to retry.
   /// </summary>
   public class RateLimitException : ApiException
   {
       public TimeSpan? RetryAfter { get; }

       public RateLimitException(string message, TimeSpan? retryAfter = null)
           : base(message, -3, 429)
       {
           RetryAfter = retryAfter;
       }
   }

   /// <summary>
   /// WebSocket connection error.
   /// </summary>
   public class ConnectionException : FugleException
   {
       public ConnectionException(string message, int errorCode = -5)
           : base(message, errorCode) { }
   }

   /// <summary>
   /// Internal error from Rust panic or unexpected failure.
   /// </summary>
   public class FugleInternalException : FugleException
   {
       public FugleInternalException(string message = "Internal error occurred")
           : base(message, -999) { }
   }
   ```

2. Create RestClient.cs with TaskCompletionSource pattern:
   ```csharp
   using System.Runtime.InteropServices;
   using System.Text.Json;
   using Fugle.MarketData.Models;

   namespace Fugle.MarketData;

   /// <summary>
   /// REST client for Fugle Market Data API.
   /// Provides both async (Task-based) and sync methods for all endpoints.
   /// </summary>
   public sealed class RestClient : IDisposable
   {
       private IntPtr _handle;
       private bool _disposed;

       public RestClient(string apiKey)
       {
           if (string.IsNullOrEmpty(apiKey))
               throw new ArgumentNullException(nameof(apiKey));

           _handle = NativeMethods.fugle_rest_client_new(apiKey);
           if (_handle == IntPtr.Zero)
               throw new FugleException("Failed to create REST client");
       }

       #region Stock Intraday

       /// <summary>
       /// Get stock quote asynchronously.
       /// </summary>
       public Task<Quote> GetStockQuoteAsync(string symbol, CancellationToken cancellationToken = default)
           => CallAsync<Quote>(
               (cb, ud) => NativeMethods.fugle_rest_stock_quote_async(_handle, symbol, cb, ud),
               cancellationToken
           );

       /// <summary>
       /// Get stock quote synchronously.
       /// </summary>
       public Quote GetStockQuote(string symbol)
           => GetStockQuoteAsync(symbol).GetAwaiter().GetResult();

       /// <summary>
       /// Get stock trades asynchronously.
       /// </summary>
       public Task<TradesResponse> GetStockTradesAsync(string symbol, CancellationToken cancellationToken = default)
           => CallAsync<TradesResponse>(
               (cb, ud) => NativeMethods.fugle_rest_stock_trades_async(_handle, symbol, cb, ud),
               cancellationToken
           );

       /// <summary>
       /// Get stock trades synchronously.
       /// </summary>
       public TradesResponse GetStockTrades(string symbol)
           => GetStockTradesAsync(symbol).GetAwaiter().GetResult();

       /// <summary>
       /// Get stock ticker asynchronously.
       /// </summary>
       public Task<Ticker> GetStockTickerAsync(string symbol, CancellationToken cancellationToken = default)
           => CallAsync<Ticker>(
               (cb, ud) => NativeMethods.fugle_rest_stock_ticker_async(_handle, symbol, cb, ud),
               cancellationToken
           );

       /// <summary>
       /// Get stock ticker synchronously.
       /// </summary>
       public Ticker GetStockTicker(string symbol)
           => GetStockTickerAsync(symbol).GetAwaiter().GetResult();

       #endregion

       #region Async Implementation

       // Delegate matching Rust callback signature
       private delegate void ResultCallback(IntPtr userData, IntPtr resultJson, int errorCode);

       private async Task<T> CallAsync<T>(
           Action<ResultCallback, IntPtr> nativeCall,
           CancellationToken cancellationToken)
       {
           ThrowIfDisposed();

           var tcs = new TaskCompletionSource<T>(TaskCreationOptions.RunContinuationsAsynchronously);

           using var registration = cancellationToken.Register(
               () => tcs.TrySetCanceled(cancellationToken));

           var handle = GCHandle.Alloc((tcs, typeof(T)));
           try
           {
               ResultCallback callback = OnResultCallback<T>;
               nativeCall(callback, GCHandle.ToIntPtr(handle));

               return await tcs.Task.ConfigureAwait(false);
           }
           catch (Exception) when (handle.IsAllocated)
           {
               handle.Free();
               throw;
           }
       }

       private static void OnResultCallback<T>(IntPtr userData, IntPtr resultJson, int errorCode)
       {
           var handle = GCHandle.FromIntPtr(userData);
           var (tcs, _) = ((TaskCompletionSource<T>, Type))handle.Target!;
           handle.Free();

           if (errorCode != NativeErrorCodes.Success)
           {
               tcs.TrySetException(NativeErrorCodes.ToException(errorCode));
               return;
           }

           try
           {
               var json = Marshal.PtrToStringUTF8(resultJson);
               NativeMethods.fugle_free_string(resultJson);

               if (string.IsNullOrEmpty(json))
               {
                   tcs.TrySetException(new FugleException("Empty response"));
                   return;
               }

               var result = JsonSerializer.Deserialize<T>(json);
               if (result is null)
               {
                   tcs.TrySetException(new FugleException("Failed to deserialize response"));
                   return;
               }

               tcs.TrySetResult(result);
           }
           catch (Exception ex)
           {
               tcs.TrySetException(new FugleException("Deserialization error", -999, ex));
           }
       }

       #endregion

       #region IDisposable

       private void ThrowIfDisposed()
       {
           if (_disposed)
               throw new ObjectDisposedException(nameof(RestClient));
       }

       public void Dispose()
       {
           if (_disposed) return;

           if (_handle != IntPtr.Zero)
           {
               NativeMethods.fugle_rest_client_free(_handle);
               _handle = IntPtr.Zero;
           }

           _disposed = true;
           GC.SuppressFinalize(this);
       }

       ~RestClient() => Dispose();

       #endregion
   }
   ```
